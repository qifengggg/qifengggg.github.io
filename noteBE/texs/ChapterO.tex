\chapter*{大作文的格式}

\begin{itemize}
    \item \textbf{第一段}
    
    描述漫画内容。

    \begin{enumerate}[label = \Alph*.]
        \item 极简主义 - 抓住与漫画主题相关的细节；
        \item 时态 - 现在进行时、一般现在时；
        \item 汉语提示 - 不要直译
    \end{enumerate}
    \item \textbf{第二段}
    
    \textbf{论证某种品质的重要性;}

    阐述现象重要性;
    \item \textbf{第三段}
    
    提出建议，说明怎么做。
\end{itemize}
